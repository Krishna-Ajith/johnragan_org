});

attr('some_attr') Setter Example

This paragraph has an attr of "some_attr". Click on this paragraph to see the output of the following code to set a new value for "some_attr".

$('#attr_setter').click( function() {
	$('#attr_setter').attr({'some_attr': 'new_value'})
	alert($('#attr_setter').attr('some_attr'));
});
